    <script>
        let currentProxy = { type: 'cors-anywhere' };

        function toggleModal(show) {
            document.getElementById('proxyModal').style.display = show ? 'flex' : 'none';
        }

        function toggleFields() {
            const type = document.getElementById('proxyType').value;
            document.getElementById('customFields').style.display = (type === 'cors-anywhere') ? 'none' : 'block';
        }

        function saveProxy() {
            currentProxy = {
                type: document.getElementById('proxyType').value,
                host: document.getElementById('proxyHost').value,
                port: document.getElementById('proxyPort').value,
                user: document.getElementById('proxyUser').value,
                pass: document.getElementById('proxyPass').value
            };
            
            document.getElementById('proxyStatus').innerText = `Active Proxy: ${currentProxy.type.toUpperCase()} ${currentProxy.host || ''}`;
            toggleModal(false);
            alert('Proxy configuration updated!');
        }

        async function navigate() {
            const url = document.getElementById('urlInput').value;
            const iframe = document.getElementById('viewport');

            if (currentProxy.type === 'cors-anywhere') {
                // Implementing CORS Anywhere logic with required headers
                const proxyUrl = 'https://cors-anywhere.herokuapp.com/';
                
                try {
                    const response = await fetch(proxyUrl + url, {
                        headers: {
                            'X-Requested-With': 'XMLHttpRequest',
                            'Origin': window.location.origin
                        }
                    });
                    const html = await response.text();
                    
                    // Injecting content into iframe
                    const blob = new Blob([html], { type: 'text/html' });
                    iframe.src = URL.createObjectURL(blob);
                } catch (err) {
                    alert("CORS Anywhere error. Make sure you visited https://cors-anywhere.herokuapp.com/corsdemo to request access.");
                }
            } else {
                // For SOCKS5/HTTP/WISP/HTTPS custom:
                // Standard browsers do not allow changing proxy settings via JS for security.
                // We simulate the intent by attempting a direct load or logging the config.
                console.log("Routing through custom proxy:", currentProxy);
                iframe.src = url;
                alert(`${currentProxy.type.toUpperCase()} selected. Note: Browsers require a browser extension or system-wide settings to route raw socket traffic.`);
            }
        }
    </script>
