    <script>
        let config = { type: 'cors-anywhere', ip: '', port: '' };

        function toggleModal() {
            const modal = document.getElementById('proxy-modal');
            const overlay = document.getElementById('overlay');
            const show = modal.style.display !== 'block';
            modal.style.display = show ? 'block' : 'none';
            overlay.style.display = show ? 'block' : 'none';
        }

        function handleTypeChange() {
            const type = document.getElementById('proxy-type').value;
            document.getElementById('manual-settings').style.display = (type === 'cors-anywhere') ? 'none' : 'block';
        }

        function saveProxy() {
            config.type = document.getElementById('proxy-type').value;
            config.ip = document.getElementById('proxy-ip').value;
            config.port = document.getElementById('proxy-port').value;
            
            const typeLabel = document.getElementById('proxy-type').options[document.getElementById('proxy-type').selectedIndex].text;
            document.getElementById('proxy-status').innerText = `Status: Using ${typeLabel}` + (config.ip ? ` (${config.ip}:${config.port})` : '');
            toggleModal();
        }

        async function navigate() {
            const targetUrl = document.getElementById('url-input').value;
            const frame = document.getElementById('content-frame');

            if (config.type === 'cors-anywhere') {
                const proxyBase = 'https://cors-anywhere.herokuapp.com/';
                
                try {
                    // Fetching through proxy with required headers
                    const response = await fetch(proxyBase + targetUrl, {
                        method: 'GET',
                        headers: {
                            'X-Requested-With': 'XMLHttpRequest', // Fixes "Missing required request header"
                            'Origin': window.location.origin      // Required for CORS handshakes
                        }
                    });
                    
                    if (!response.ok) throw new Error();
                    const html = await response.text();
                    
                    const doc = frame.contentWindow.document;
                    doc.open();
                    doc.write(html);
                    doc.close();
                } catch (e) {
                    alert('Error: Please ensure you have requested temporary access at https://cors-anywhere.herokuapp.com/corsdemo first.');
                }
            } else {
                // Browsers do not allow raw TCP/Proxy tunneling via client-side JS alone
                alert(`Routing via ${config.type} at ${config.ip}:${config.port}. (Simulated)`);
                frame.src = targetUrl;
            }
        }
    </script>
